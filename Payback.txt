    <script>
        // Function to calculate Payback Period
        function calculatePayback() {
            // Get the input values
            let initialInvestment = document.getElementById('initial-investment').value;
            let annualCashInflows = document.getElementById('annual-cash-inflows').value;
            let result = document.getElementById('result');
            let steps = document.getElementById('calculation-steps');
            let comments = document.getElementById('comments');

            // Check if inputs are valid
            if (initialInvestment && annualCashInflows && annualCashInflows > 0) {
                // Calculate the payback period
                let paybackPeriod = initialInvestment / annualCashInflows;
                result.innerHTML = `<strong>Payback Period: ${paybackPeriod.toFixed(2)} years</strong>`;

                // Show the calculation steps
                steps.innerHTML = `
                    <p><strong>Calculation Steps:</strong></p>
                    <p>1. Initial Investment: LKR ${initialInvestment}</p>
                    <p>2. Annual Cash Inflows: LKR ${annualCashInflows}</p>
                    <p>3. Payback Period = Initial Investment / Annual Cash Inflows</p>
                    <p>4. Payback Period = ${initialInvestment} / ${annualCashInflows} = <strong>${paybackPeriod.toFixed(2)} years</strong></p>
                `;

                // Provide interpretation of the payback period result
                if (paybackPeriod < 3) {
                    comments.innerHTML = "This is a quick return on investment! The payback period is considered short.";
                } else if (paybackPeriod >= 3 && paybackPeriod <= 5) {
                    comments.innerHTML = "This is a moderate return on investment. The payback period is acceptable for most investments.";
                } else {
                    comments.innerHTML = "This is a slow return on investment. It may carry higher liquidity risk.";
                }
            } else {
                // If inputs are invalid or empty
                result.innerHTML = '';
                steps.innerHTML = '';
                comments.innerHTML = 'Please enter valid values for both fields.';
            }
        }
    </script>
