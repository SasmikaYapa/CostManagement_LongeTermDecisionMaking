    <script>
        // Function to calculate ROI and show the calculation steps
        function calculateROI() {
            // Get values from input fields
            let netProfit = document.getElementById('net-profit').value;
            let investmentCost = document.getElementById('investment-cost').value;
            let result = document.getElementById('result');
            let steps = document.getElementById('calculation-steps');
            let comments = document.getElementById('comments');

            // Ensure both values are entered and valid
            if (netProfit && investmentCost && investmentCost > 0) {
                // Calculate ROI
                let roi = (netProfit / investmentCost) * 100;
                result.innerHTML = `<strong>ROI: ${roi.toFixed(2)}%</strong>`; // Display the ROI percentage

                // Display the calculation steps
                steps.innerHTML = `
                    <p><strong>Calculation Steps:</strong></p>
                    <p>1. Net Profit: LKR ${netProfit}</p>
                    <p>2. Cost of Investment: LKR ${investmentCost}</p>
                    <p>3. ROI = (Net Profit / Cost of Investment) × 100</p>
                    <p>4. ROI = (${netProfit} / ${investmentCost}) × 100 = <strong>${roi.toFixed(2)}%</strong></p>
                `;

                // Provide comments based on ROI value
                if (roi > 100) {
                    comments.innerHTML = "Excellent return! This investment is highly profitable.";
                } else if (roi > 50) {
                    comments.innerHTML = "Good return! The investment is performing well.";
                } else if (roi > 0) {
                    comments.innerHTML = "Moderate return. The investment is making a profit.";
                } else if (roi == 0) {
                    comments.innerHTML = "No return. You're breaking even on this investment.";
                } else {
                    comments.innerHTML = "Loss! This investment is losing money.";
                }
            } else {
                // If inputs are invalid or empty
                result.innerHTML = '';
                steps.innerHTML = '';
                comments.innerHTML = 'Please enter valid values for both fields.';
            }
        }
    </script>
