<script>
    function calculateCVP() {
        let salesPrice = parseFloat(document.getElementById('salesPrice').value);
        let variableCost = parseFloat(document.getElementById('variableCost').value);
        let fixedCost = parseFloat(document.getElementById('fixedCost').value);
        let budgetedSales = parseFloat(document.getElementById('budgetedSales').value);
        let urgency = document.getElementById('urgency').value;

        if(isNaN(salesPrice) || isNaN(variableCost) || isNaN(fixedCost) || isNaN(budgetedSales)) {
            alert("Please enter valid numbers in all fields.");
            return;
        }

        // Contribution per unit
        let contribution = salesPrice - variableCost;

        // BEP in units
        let bepUnits = fixedCost / contribution;

        // BEP in sales value
        let bepSalesValue = bepUnits * salesPrice;

        // Margin of Safety
        let mos = (budgetedSales - bepUnits) * salesPrice;

        // Display results with urgency
        let resultDiv = document.getElementById('result');
        resultDiv.innerHTML = `
            <h3>Urgency Level: ${urgency.charAt(0).toUpperCase() + urgency.slice(1)}</h3>
            <h3>Contribution per Unit: Rs. ${contribution.toFixed(2)}</h3>
            <h3>Break-Even Point (Units): ${bepUnits.toFixed(2)} units</h3>
            <h3>Break-Even Point (Sales Value): Rs. ${bepSalesValue.toFixed(2)}</h3>
            <h3>Margin of Safety: Rs. ${mos.toFixed(2)}</h3>
        `;
    }

    function exportPDF() {
        const { jsPDF } = window.jspdf;
        const doc = new jsPDF();
        const resultContent = document.getElementById('result').innerText;

        if (resultContent === "") {
            alert("Please calculate the CVP first before downloading the results.");
            return;
        }

        doc.text(resultContent, 10, 10);
        doc.save("CVP_results.pdf");
    }
</script>
