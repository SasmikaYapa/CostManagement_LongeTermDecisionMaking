        <!-- Description of DCF -->
        <div class="description">
            <p><strong>Discounted Cash Flow (DCF)</strong> analysis is a method used to value an investment based on its expected future cash flows. By estimating these cash flows for each period and discounting them back to their present value using a discount rate, investors can evaluate the attractiveness of an investment opportunity.</p>
            <p>The formula for DCF is as follows:</p>
            <p><strong>DCF = Cash Flow / (1 + Discount Rate)^t</strong></p>
            <p>The DCF model is embedded in the calculations of NPV and IRR, providing insights into the financial viability of short-term and long-term projects.</p>
        </div>

        <!-- Input fields for Initial Investment, Discount Rate, and dynamic Cash Flows -->

    <script>
        // Function to add a new cash flow input field
        function addCashFlow() {
            const cashFlowsContainer = document.getElementById('cash-flows-container');
            const cashFlowGroup = document.createElement('div');
            cashFlowGroup.className = 'cash-flow-group';
            cashFlowGroup.innerHTML = `
                <input type="number" class="cash-flow-input" placeholder="Enter cash flow" required>
                <button class="remove-btn" onclick="removeCashFlow(this)">Remove</button>
            `;
            cashFlowsContainer.appendChild(cashFlowGroup);
        }

        // Function to remove a cash flow input field
        function removeCashFlow(button) {
            const cashFlowGroup = button.parentElement;
            cashFlowGroup.remove();
        }

        // Function to calculate DCF, NPV, and IRR
        function calculateDCF() {
            let discountRate = parseFloat(document.getElementById('discount-rate').value) / 100;
            let cashFlowInputs = document.querySelectorAll('.cash-flow-input');
            let result = document.getElementById('result');
            let steps = document.getElementById('calculation-steps');
            let comments = document.getElementById('comments');

            // Parse the cash flows input
            let cashFlows = Array.from(cashFlowInputs).map(input => parseFloat(input.value));

            // Validate inputs
            if (isNaN(discountRate) || cashFlows.some(isNaN)) {
                result.innerHTML = "Please enter valid numbers for all fields.";
                steps.innerHTML = '';
                comments.innerHTML = '';
                return;
            }

            // DCF Calculation
            let dcf = 0;
            for (let t = 0; t < cashFlows.length; t++) {
                dcf += cashFlows[t] / Math.pow(1 + discountRate, t + 1);
            }

            // NPV Calculation (Assuming an initial investment of zero for demonstration)
            let npv = dcf;

            // Interpretation and Display
            result.innerHTML = `
                <strong>DCF: LKR ${dcf.toFixed(2)}</strong><br>
                <strong>NPV (assuming initial investment of 0): LKR ${npv.toFixed(2)}</strong>
            `;

            steps.innerHTML = `
                <p><strong>Calculation Steps:</strong></p>
                <p>1. DCF = Î£ (Cash Flow / (1 + Discount Rate)^t)</p>
                <p>2. NPV = DCF - Initial Investment (0 in this case)</p>
            `;

            // Interpretation based on DCF
            if (dcf > 0) {
                comments.innerHTML = "The investment is attractive as the DCF is positive.";
            } else if (dcf === 0) {
                comments.innerHTML = "The investment breaks even as the DCF is zero.";
            } else {
                comments.innerHTML = "The investment is not attractive as the DCF is negative.";
            }
        }
    </script>
